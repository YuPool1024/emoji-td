```js
// 成就系统
(function(){
  const ACHIEVEMENTS = [
    { id:'first_win', name:'首次通关', cond:'任意难度通关', emoji:'🏆', check:g=>g && g.state==='won' },
    { id:'hard_clear', name:'硬核征服', cond:'硬核通关', emoji:'🥇', check:g=>g && g.state==='won' && g.diff==='hard' },
    { id:'tactic_air', name:'制空', cond:'一局击杀 ≥30 飞行怪', emoji:'✈️', check:g=>g && g.airKills>=30 },
    { id:'tactic_armor', name:'破甲', cond:'一局击杀 ≥20 重甲怪', emoji:'🛡️', check:g=>g && g.armorKills>=20 },
    { id:'full_comp', name:'全职业', cond:'一局集齐 6 种塔', emoji:'🌟', check:g=>{
        if(!g)return false; const s=new Set(g.towerBuildHistory); return s.size>=6;
    }},
    { id:'completionist', name:'集邮', cond:'解锁全部塔的 tier-3', emoji:'💎', check:g=>{
        if(!g)return false; const tier3=new Set(g.towers.filter(t=>t.level>=3).map(t=>t.type)); return tier3.size>=6;
    }},
    { id:'pacifist', name:'零漏', cond:'通关且漏怪 = 0', emoji:'🎯', check:g=>g && g.state==='won' && g.leaks===0 },
    { id:'speedrun', name:'神速', cond:'硬核通关 ≤ 8 分钟', emoji:'⚡', check:g=>g && g.state==='won' && g.diff==='hard' && g.duration<=480 },
    { id:'tower_of_life', name:'不死之塔', cond:'通关且英雄未死', emoji:'❤️', check:g=>{
        if(!g||g.state!=='won')return false; return g.hero && g.hero.alive;
    }},
  ];

  function loadAchievements(){
    try{ return JSON.parse(localStorage['td_achievements'])||[]; }catch(e){ return []; }
  }
  function saveAchievements(list){
    localStorage['td_achievements'] = JSON.stringify(list);
  }

  // 检查解锁（在 game end 时调用）
  function checkUnlocks(g){
    const unlocked = loadAchievements();
    let newOnes = [];
    for (const a of ACHIEVEMENTS){
      if (!unlocked.includes(a.id) && a.check(g)){
        unlocked.push(a.id);
        newOnes.push(a);
      }
    }
    if (newOnes.length) saveAchievements(unlocked);
    return newOnes;
  }

  function createAchievementsPanel(){
    let parent = null;
    function show(){
      if(!parent)return;
      const unlocked = loadAchievements();
      let html = '<h1>🏆 成就</h1><div class="ach-list">';
      for (const a of ACHIEVEMENTS){
        const done = unlocked.includes(a.id);
        html += '<div class="ach-item'+(done?' ach-done':' ach-locked')+'">'+
          '<span class="ach-emoji">'+(done?a.emoji:'🔒')+'</span>'+
          '<span class="ach-name">'+a.name+'</span>'+
          '<span class="ach-cond">'+a.cond+'</span>'+
          '</div>';
      }
      html += '</div><div class="end-btns"><button onclick="location.reload()">返回</button></div>';
      parent.className = 'overlay show';
      parent.innerHTML = html;
    }
    return {
      mount(parentEl){ parent = parentEl; },
      show,
      checkUnlocks,
      getProgress(){ return loadAchievements().length+'/'+ACHIEVEMENTS.length; },
    };
  }

  if(typeof module!=='undefined') module.exports = { createAchievementsPanel, ACHIEVEMENTS, checkUnlocks };
  else window.createAchievementsPanel = createAchievementsPanel;
})();
```

```js
const HERO_TYPES = {
  warrior: { emoji:'🦸', hp:120, maxHp:120, radius:1.6, stickCount:2, dps:30, reviveCost:60, unlock:null },
  mage:    { emoji:'🧙', hp:80,  maxHp:80,  radius:2.0, stickCount:0, dps:20, reviveCost:80, unlock:'full_comp', dot:10 },
  hunter:  { emoji:'🏹', hp:100, maxHp:100, radius:4.0, stickCount:2, dps:50, reviveCost:100, unlock:'pacifist', firstHitMul:1.5 },
};

function makeHero(type, r, c){
  const h = HERO_TYPES[type] || HERO_TYPES.warrior;
  return { type:type||'warrior', r, c, ...h, level:1, alive:true, attackCd:0, reviveTimer:-1, hp:h.hp };
}
```
